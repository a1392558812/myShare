# css+js大屏适配

## 方案一：rem适配

### 定义设计稿基准

```javascript
const DESIGN_WIDTH = 1920;
const BASE_REM = 1; // 1rem = 设计稿1px

export const initRemAdapt = () => {
  const setRootFontSize = () => {
    const winW = document.documentElement.clientWidth || window.innerWidth;
    // winW / DESIGN_WIDTH = winFont / BASE_REM

    /**
     * TODO: 奇怪屏幕比例，额外处理
     * 例如：当 winW <= 1200 时候，则一律按 winW = 1200 处理
     *
     */
    const fontSize = (winW / DESIGN_WIDTH) * BASE_REM;
    document.documentElement.style.fontSize = `${fontSize}px`;
    document.body.style.fontSize = "14px"; // 用body字体大小，避免元素默认字体大小被修改
  };

  setRootFontSize();

  let timer = null;
  window.addEventListener("resize", () => {
    clearTimeout(timer);
    timer = setTimeout(setRootFontSize, 100);
  });
};

export const px2rem = (px) => {
  return `${px / BASE_REM}rem`;
};
```

### 使用示例

// main.js

```javascript
import { initRemAdapt } from "./style-utils";
initRemAdapt();
```

```html
<template>
  <div ref="box1Ref" class="box1"></div>
  <div class="box2"></div>
</template>

<script setup lang="ts">
  import { px2rem } from "./style-utils";

  const box1Ref = ref(null);
  onMounted(() => {
    // 设计稿 1920 下 100px
    box1Ref.value.style.width = px2rem(100);
    box1Ref.value.style.height = px2rem(100);
  });
</script>

<style scoped lang="scss">
  .box2 {
    width: 200rem;
    height: 200rem;
    background-color: blue;
  }
</style>
```

## 方案二：vw/vh适配

### 定义设计稿基准

// variables.scss

```scss
$design-width: 1920;
$design-height: 1080;
@use "sass:math";

@function vw($px) {
  $px: $px * 1px;
  @return math.div($px, $design-width) * 100vw;
}

@function vh($px) {
  $px: $px * 1px;
  @return math.div($px, $design-height) * 100vh;
}

@function size($width, $height: null) {
  @if $height == null {
    @return vw($width);
  }
  @return vw($width) vh($height);
}

// 最小字体大小
@function responsive-font($min, $max, $px) {
  $vw: math.div($px, $design-width) * 100;
  @return clamp(#{$min}px, #{$vw}vw, #{$max}px);
}
```

### js方法

```javascript
export const DESIGN_WIDTH = 1920;
export const DESIGN_HEIGHT = 1080;
export interface ResizeObserverOptions {
  beforeCallback?: () => void;
  callback?: () => void;
  afterCallback?: () => void;
  delay?: number;
}

export const px2vw = (px) => {
  const vwValue = (px / DESIGN_WIDTH) * 100;
  /**
   * TODO: 极端大/小屏或者奇怪比例带鱼屏的情况，需要特殊处理
   * 例如：
   * if (window.innerWidth > 3840) {
   *  return `${Math.min(vwValue, 80)}vw`;
   * }
   * if (window.innerWidth < 1366) {
   *  return `${Math.max(vwValue, 5)}vw`;
   * }
   */
  return `${vwValue}vw`;
};

export const px2vh = (px) => {
  const vhValue = (px / DESIGN_HEIGHT) * 100;
  /**
   * TODO: 极端大/小屏或者奇怪比例带鱼屏的情况，需要特殊处理
   * 例如：
   * if (window.innerHeight > 2160) {
   *  return `${Math.min(vhValue, 80)}vh`;
   * }
   * if (window.innerHeight < 768.375) {
   *  return `${Math.max(vhValue, 5)}vh`;
   * }
   */
  return `${vhValue}vh`;
};

export const applyStyles = (element, styles) => {
  Object.keys(styles).forEach((key) => {
    const value = styles[key];

    if (typeof value === "number") {
      if (
        key.includes("width") ||
        key.includes("left") ||
        key.includes("right")
      ) {
        element.style[key] = px2vw(value);
      } else if (
        key.includes("height") ||
        key.includes("top") ||
        key.includes("bottom")
      ) {
        element.style[key] = px2vh(value);
      } else {
        element.style[key] = px2vh(value);
      }
    }
  });
};

export const initResizeObserver = (options = {
  beforeCallback,
  callback,
  afterCallback,
  delay = 200,
}: ResizeObserverOptions) => {
  let resizeTimer;
  const handleResize = () => {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(() => {
      if (options.beforeCallback) {
        // 自定义前置逻辑
        options.beforeCallback();
      }

      /**
       * TODO： 前置公用逻辑
       */

      if (options.callback) {
        // 自定义逻辑
        options.callback();
      }

      /**
       * TODO： 后置公用逻辑
       */

      if (options.afterCallback) {
        // 自定义后置逻辑
        options.afterCallback();
      }

      console.log(
        "屏幕尺寸变化:",
        window.innerWidth + "×" + window.innerHeight,
      );
    }, options.delay);
  };

  window.addEventListener("resize", handleResize);

  return () => {
    window.removeEventListener("resize", handleResize);
  };
};
```

### 使用示例

```html
<template>
  <div ref="box1Ref" class="box1">
    <div class="custom-content"></div>
  </div>
  <div ref="box2Ref" class="box2"></div>
</template>

<script setup lang="ts">
  import { applyStyles, px2vw, px2vh, initResizeObserver } from "@/utils/style-utils";
  const box1Ref = useTemplateRef("box1Ref");
  const box2Ref = useTemplateRef("box2Ref");

  onMounted(() => {
    applyStyles(box1Ref.value!, {
      position: 'absolute',
      left: 114514,
      top: 114514,
      width: 100,
      height: 100,
      fontSize: 16,
      padding: 20,
      '--custom-width': px2vw(114514),
      '--custom-height': px2vh(114514)
    });

    box2Ref.value!.style.width = px2vw(200);
    box2Ref.value!.style.height = px2vh(150);

    const stopResizeObserver = initResizeObserver({
      callback: () => {
        console.log('屏幕尺寸变化');
        setTimeout(() => {
          stopResizeObserver();
        }, 2000);
      }
    });
  });
</script>

<style lang="scss" scoped>
.box1 {

    height: var(--custom-height);
    background-color: red;
  }

  width: vw(150);
  height: auto;
  aspect-ratio: 16/9;
  font-size: responsive-font(12, 14, 14);
}
```

## scale 方案

- 注意：
  - 文字发虚： 缩放比例不是整数时，渲染时会文字发虚。解决办法：给文字容器单独设置 will-change: transform 或者用 -webkit-font-smoothing: antialiased，（不一定好用）。
  - 鼠标坐标偏移： scale 缩放后，DOM 元素的实际位置和视觉位置不一致。如果大屏上有 tooltip、弹窗、拖拽等交互，鼠标位置会对不上。在 ECharts 的 tooltip 上尤为明显。
  - 超宽屏留白： 例如16:9 的设计稿放到 32:9 的拼接屏上

```javascript
export const DESIGN_WIDTH = 1920;
export const DESIGN_HEIGHT = 1080;

export const initScaleAdapt = (wrapSelector = "#app-wrap") => {
  const setScale = () => {
    const winW = document.documentElement.clientWidth;
    const winH = document.documentElement.clientHeight;

    const scaleW = winW / DESIGN_WIDTH;
    const scaleH = winH / DESIGN_HEIGHT;

    const scale = Math.min(scaleW, scaleH);

    const wrap = document.querySelector(wrapSelector);
    if (!wrap) return;

    wrap.style.width = `${DESIGN_WIDTH}px`;
    wrap.style.height = `${DESIGN_HEIGHT}px`;
    wrap.style.transform = `scale(${scale})`;
    wrap.style.transformOrigin = "left top";

    const marginLeft = (window.innerWidth - DESIGN_WIDTH * scale) / 2;
    const marginTop = (window.innerHeight - DESIGN_HEIGHT * scale) / 2;

    wrap.style.marginLeft = marginLeft + "px";
    wrap.style.marginTop = marginTop + "px";
  };

  setScale();

  let timer = null;
  window.addEventListener("resize", () => {
    clearTimeout(timer);
    timer = setTimeout(setScale, 120);
  });
};
```

## 其他方案概况

|   方案   |         适合场景         |           不适合场景           |
| :------: | :----------------------: | :----------------------------: |
|  scale   |   比例固定的展示型大屏   |     超宽屏/非标比例/有交互     |
|  vw/vh   | 需要铺满全屏的响应式大屏 |      ECharts 字体适配麻烦      |
|   rem    |           都行           | 没有，就是使用麻烦点，需要换算 |
| 混合方案 |           都行           |   小 demo 用不着，纯自己别扭   |
