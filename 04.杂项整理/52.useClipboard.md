# useClipboard

> 社区中已经有相当成熟的实现，这里仅作为小项目不想引入一些大型的库时候的替代。
> [@vueuse/core useClipboard](https://vueuse.nodejs.cn/core/useClipboard/#useclipboard)

## 定义hook

```typescript
import { ref } from "vue";

export const useClipboard = (delay = 1500) => {
  const isCopied = ref(false);
  const error = ref("");
  let timer: number | null = null;

  const handleSuccess = () => {
    isCopied.value = true;
    error.value = "";
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => (isCopied.value = false), delay);
    return true;
  };

  const handleError = (err: any) => {
    error.value = err?.message || "复制失败";
    console.error("复制失败:", err);
    return false;
  };

  const copyText = async (text: string) => {
    try {
      await navigator.clipboard.writeText(text);
      return handleSuccess();
    } catch (err) {
      return handleError(err);
    }
  };

  // 复制富文本 一般富文本编辑器会用到这个方法，使用copyRichText和copyText分别复制同一段富文本粘贴到QQ或者微信或者富文本编辑器中，会显示不同的内容
  /*
    example:
    // 复制富文本
    const htmlContent = '<div style="color: red; font-size: 12px;">这是一个段落<img src="xxxxx" /><span style="color: blue; font-size: 20px;">这是一个span标签</span></div>'
    await copyRichText(htmlContent)
    await copyText(htmlContent)
    // 粘贴到QQ或者微信会显示为段落
  */
  const copyRichText = async (htmlStr: string) => {
    try {
      const blobHtml = new Blob([htmlStr], { type: "text/html" });
      const item = new ClipboardItem({
        "text/html": blobHtml,
      });
      await navigator.clipboard.write([item]);
      handleSuccess();
      return true;
    } catch (err) {
      return handleError(err);
    }
  };

  // 复制文件（大都不支持多个文件）
  const copyFiles = async (files: File[]) => {
    try {
      // 需确保浏览器支持当前文件的MIME type才能复制成功
      const clipboardItem = files.map(
        (file) =>
          new ClipboardItem({
            [file.type]: file,
          }),
      );
      await navigator.clipboard.write(clipboardItem);
      return handleSuccess();
    } catch (err) {
      return handleError(err);
    }
  };

  // 从URL复制文件
  const copyFileFromUrl = async (url: string, fileName = "clip-file") => {
    try {
      const res = await fetch(url);
      if (!res.ok) throw new Error("资源请求失败");
      const blob = await res.blob();
      const file = new File([blob], fileName, { type: blob.type });
      return await copyFiles([file]);
    } catch (err: any) {
      return handleError(err);
    }
  };

  return {
    isCopied,
    error,
    copyRichText,
    copyText,
    copyFiles,
    copyFileFromUrl,
  };
};
```

## 使用示例

```html
<template>
  <div>
    <div>
      <input v-model="text" />
      <button @click="copyText(text)">
        {{ isCopied ? 'copied' : 'copy' }}
      </button>
    </div>

    <div>
      <input type="file" multiple @change="onChangeFile" />
      <button @click="onCopyFile">{{ isCopied ? 'copied' : 'copy' }}</button>
    </div>

    <div>
      <input v-model="url" />
      <button @click="copyFileFromUrl(url)">
        {{ isCopied ? 'copied' : 'copy' }}
      </button>
    </div>

    <div>
      <input v-model="htmlContent" />
      <button @click="copyRichText(htmlContent)">
        {{ isCopied ? 'copied' : 'copy' }}
      </button>
    </div>

    <div>error: {{ error || '--' }}</div>
  </div>
</template>

<script setup lang="ts">
  import { ref } from "vue";
  import { useClipboard } from "@/hooks/useClipboard";

  const {
    isCopied,
    error,
    copyText,
    copyFiles,
    copyFileFromUrl,
    copyRichText,
  } = useClipboard();

  const text = ref("hello world");
  const files = ref<File[]>([]);
  const url = ref("https://xxxxxxxxxxx.png");
  const htmlContent = ref(`<div style="color:red;font-size:16px;">
我是红色富文本
<img src="${url.value}" alt="图片" />
文本内容
</div>`);

  const onChangeFile = (e: Event) => {
    const targetFiles = (e.target as HTMLInputElement).files || ([] as File[]);
    console.log("onChangeFile", targetFiles);
    files.value = Array.from(targetFiles);
  };

  const onCopyFile = () => {
    if (files.value.length) {
      copyFiles(files.value);
    }
  };
</script>

<style scoped></style>
```
