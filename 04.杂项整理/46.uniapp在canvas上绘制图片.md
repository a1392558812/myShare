# uniapp在canvas上绘制图片

- 部分工具函数已经抽取，本人人懒，推荐先看工具函数有无重复代码
  - @/utils/components-tools.ts ---> 04.杂项整理/35.components-tools.md

> 关于微信接口迁移：[【旧版 Canvas 迁移指南】](https://developers.weixin.qq.com/miniprogram/dev/framework/ability/canvas-legacy-migration.html)
> canvas需要注意屏幕设备像素比

```html
<template>
  <view class="">
    <view class="flex">
      <uv-input v-model="src" placeholder="请输入图片地址" />
      <uv-button @click="setInit(src)"> 绘制 </uv-button>
    </view>

    <canvas
      v-if="canvasState.cssWidth && canvasState.cssHeight"
      :id="canvasState.canvasId"
      ref="canvasRef"
      :hidpi="true"
      :canvas-id="canvasState.canvasId"
      type="2d"
      :width="canvasState.realWidth"
      :height="canvasState.realHeight"
      :style="{ width: `${canvasState.cssWidth}px`, height: `${canvasState.cssHeight}px` }"
    />
  </view>
</template>

<script setup lang="ts">
  import { getRect } from "@/utils/components-tools";
  import { getCurrentInstance, nextTick, reactive, ref } from "vue";

  const src = ref("https://picsum.photos/id/1/300/450");
  const initCanvasState = (): {
    cssWidth: number;
    cssHeight: number;
    realWidth: number;
    realHeight: number;
    canvasId: string;
    ctx: UniApp.CanvasContext | null;
  } => ({
    cssWidth: 0,
    cssHeight: 0,
    realWidth: 0,
    realHeight: 0,
    canvasId: "myCanvas",
    ctx: null,
  });

  const canvasState = reactive(initCanvasState());

  const canvasRef = ref<HTMLCanvasElement>();
  const instance = getCurrentInstance() as any;
  const pixelRatio = ref(
    (() => {
      // #ifdef MP-WEIXIN
      return uni.getWindowInfo().pixelRatio;
      // #endif

      // #ifndef MP-WEIXIN
      return 1;
      // #endif
    })(),
  );

  let canvasInstance: any;

  const setCssSize = (width: number, height: number) => {
    canvasState.realWidth = width * pixelRatio.value;
    canvasState.realHeight = height * pixelRatio.value;
    canvasState.cssWidth = width;
    canvasState.cssHeight = height;

    // #ifdef MP-WEIXIN
    if (canvasInstance) {
      canvasInstance.width = canvasState.realWidth;
      canvasInstance.height = canvasState.realHeight;
    }
    // #endif
  };

  // #ifdef MP-WEIXIN
  const canvas2dAdapter = (
    ctx: CanvasRenderingContext2D,
  ): UniApp.CanvasContext => {
    const ctxDrawImage = ctx.drawImage.bind(ctx);
    return Object.assign(ctx, {
      // 统一ctx用法，再此处拓展微信的方法
      drawImage: (...args: Parameters<UniApp.CanvasContext["drawImage"]>) => {
        return new Promise((resolve, reject) => {
          const image = canvasInstance.createImage();
          image.src = args[0];
          const newArgs = [image, ...args.slice(1)] as Parameters<
            CanvasRenderingContext2D["drawImage"]
          >;
          console.log("args", newArgs);
          image.onload = () => {
            ctxDrawImage(...newArgs);
            resolve(null);
          };
          image.onerror = (err: any) => {
            reject(new Error("drawImage error", err));
          };
        });
      },
      draw: () => {},
      /**
       * 举例：这几个方法在微信中是canvas实例的属性，
       * setMiterLimit(miterLimit: number) {
       *   ctx.miterLimit = miterLimit
       * },
       * setShadow(offsetX: number, offsetY: number, blur: number, color: string) {
       *   ctx.shadowOffsetX = offsetX
       *   ctx.shadowOffsetY = offsetY
       *   ctx.shadowBlur = blur
       *   ctx.shadowColor = color
       * },
       */
    }) as unknown as UniApp.CanvasContext;
  };
  // #endif

  const getContext = () => {
    return new Promise<UniApp.CanvasContext>((resolve) => {
      const { ctx } = canvasState;
      if (ctx) {
        return resolve(ctx);
      }
      // #ifndef MP-WEIXIN
      canvasState.ctx = uni.createCanvasContext(
        `${canvasState.canvasId}`,
        instance.proxy,
      );
      resolve(canvasState.ctx);
      // #endif

      // #ifdef MP-WEIXIN
      getRect(`#${canvasState.canvasId}`, false, instance.proxy, true).then(
        (canvasRect: any) => {
          if (canvasRect.node) {
            canvasInstance = canvasRect.node;
            canvasState.ctx = canvas2dAdapter(
              canvasInstance.getContext("2d") as CanvasRenderingContext2D,
            );
            resolve(canvasState.ctx);
          }
        },
      );
      // #endif
    });
  };

  const setInit = (src: string) => {
    // 部分平台设定好宽高canvas的宽高不可再修改，所以宽高需要在初始化时候去赋值，
    console.log("src", src);
    Object.assign(canvasState, initCanvasState());
    uni.getImageInfo({
      src,
      success: (res) => {
        canvasState.cssWidth = res.width;
        canvasState.cssHeight = res.height;
        nextTick()
          .then(() => {
            return getContext();
          })
          .then(async (ctx) => {
            setCssSize(res.width, res.height);
            ctx.scale(pixelRatio.value, pixelRatio.value);
            ctx.clearRect(0, 0, canvasState.realWidth, canvasState.realHeight);
            await ctx.drawImage(
              "https://picsum.photos/id/1/300/450",
              0,
              0,
              canvasState.cssWidth,
              canvasState.cssHeight,
            );
            ctx.draw();
          });
      },
    });
  };
</script>
```
