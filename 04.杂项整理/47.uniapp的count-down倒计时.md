# uniapp的count-down倒计时

## 使用

```html
<template>
  <view>
    <button @click="start">开始</button>
    <button @click="stop">停止</button>
    <button @click="reset">重置</button>
    <CountDown ref="countDownRef" :time="1000 * 30" :millisecond="false">
      <template #default="{ remainTime, remainTimeDate }">
        <view>
          <view>{{ remainTime }}</view>
          <view>
            <view>{{ remainTimeDate.day }}天</view>
            <view>
              {{ remainTimeDate.hour }}:{{ remainTimeDate.minute }}:{{
              remainTimeDate.second }}:{{ remainTimeDate.millisecond }}
            </view>
          </view>
        </view>
      </template>
    </CountDown>
  </view>
</template>

<script setup lang="ts">
  import CountDown from "@/components/count-down/index.vue";
  import dayjs from "dayjs";

  const countDownRef = ref<InstanceType<typeof CountDown>>(null);

  const start = () => countDownRef.value?.start();
  const stop = () => countDownRef.value?.stop();
  const reset = () => countDownRef.value?.reset();
</script>
```

## index.vue

```html
<template>
  <view :style="customStyle">
    <slot :remain-time="remainTime" :remain-time-date="remainTimeDate" />
  </view>
</template>

<script setup lang="ts">
  import { computed, onMounted, ref, watch } from "vue";

  defineOptions({
    addGlobalClass: true,
    virtualHost: true,
    styleIsolation: "shared",
  });

  const props = withDefaults(
    defineProps<{
      time?: number;
      autoStart?: boolean;
      millisecond?: boolean;
      customStyle?: AnyStyleType;
    }>(),
    {
      time: 0,
      autoStart: true,
      millisecond: false,
      customStyle: () => ({}),
    },
  );

  const emit = defineEmits(["change", "finish"]);

  let timer = -114514;
  const runing = ref(false);
  // 结束的毫秒时间戳
  const endTime = ref(0);
  // 剩余的毫秒时间
  const remainTime = ref(0);

  const remainTimeDate = computed(() => {
    const day = Math.floor(remainTime.value / (1000 * 60 * 60 * 24));
    const hour = Math.floor(remainTime.value / (1000 * 60 * 60)) % 24;
    const minute = Math.floor(remainTime.value / (1000 * 60)) % 60;
    const second = Math.floor(remainTime.value / 1000) % 60;
    const millisecond = Math.floor(remainTime.value % 1000);

    return {
      day,
      hour,
      minute,
      second,
      millisecond,
    };
  });

  const clearTimeoutFun = () => {
    clearTimeout(timer);
    timer = -114514;
  };

  const pause = () => {
    runing.value = false;
    clearTimeoutFun();
  };

  // 设置剩余的时间
  const setRemainTime = (remain: number) => {
    remainTime.value = remain;
    emit("change", remainTime.value);
    // 如果时间已到，停止倒计时
    if (remain <= 0) {
      pause();
      emit("finish");
    }
  };

  const getRemainTime = () => {
    return Math.max(endTime.value - Date.now(), 0);
  };

  // 秒级倒计时
  const macroTick = () => {
    clearTimeoutFun();
    timer = setTimeout(() => {
      // 获取精准的剩余毫秒数
      const remain = getRemainTime();
      // 粗略估计秒数相等
      if (
        !(Math.floor(remain / 1000) === Math.floor(remainTime.value / 1000)) ||
        remain === 0
      ) {
        setRemainTime(remain);
      }
      if (remainTime.value !== 0) {
        macroTick();
      }
    }, 30);
  };

  // 毫秒级倒计时
  const microTick = () => {
    clearTimeoutFun();
    timer = setTimeout(() => {
      setRemainTime(getRemainTime());
      if (remainTime.value !== 0) {
        microTick();
      }
    }, 50);
  };

  const toTick = () => {
    if (props.millisecond) {
      return microTick();
    }
    return macroTick();
  };

  const start = () => {
    if (runing.value) return;
    runing.value = true;
    endTime.value = Date.now() + remainTime.value;
    toTick();
  };

  const reset = () => {
    pause();
    remainTime.value = props.time;
    setRemainTime(remainTime.value);
    if (props.autoStart) {
      start();
    }
  };

  onMounted(() => {
    reset();
  });

  watch(
    () => props.time,
    () => {
      reset();
    },
  );

  defineExpose({
    start,
    pause,
    reset,
  });
</script>
```
